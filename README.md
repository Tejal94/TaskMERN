# GETTING STARTED --------------

1)
        cd client              
        npm install           
        npm start       

2)
        cd server           
        npm install             
        npm start                        

# DEPENDENCIES ------------------

    FRONT END:
        react-router-dom
        react-hot-toast
        formik
        tailwind css
        react-bootstrap
        axios
        autoprefixer
        postcss

    BACKEND:
        mongoose
        bcrypt
        bcryptjs
        cors
        express
        jsonwebtoken
        morgen
        multer
